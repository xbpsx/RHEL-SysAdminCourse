## **Циклы для автоматизации**

```bash
# Цикл по всем .conf файлам
for config_file in /etc/nginx/sites-available/*.conf; do
    echo "Проверяю: $config_file"
    nginx -t -c "$config_file" || echo "Ошибка в $config_file"
done

# Цикл по серверам в кластере
servers=("web1" "web2" "web3" "db1")
for server in "${servers[@]}"; do
    ssh "admin@$server" "systemctl restart app-service"
done

# Бесконечный цикл мониторинга
while true; do
    if ! curl -f http://localhost:8080/health > /dev/null 2>&1; then
        echo "Сервис упал! Перезапускаю..."
        systemctl restart app-service
    fi
    sleep 30
done
```
